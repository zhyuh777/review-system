% (a) R竖支 VD顶横→右 6V(+上)竖支: VD阴极接6V+ → 反偏
\begin{circuitikz}[scale=1.2,transform shape]
  \draw (0,0) to[R,l=$R$] (0,3.5);
  \draw (0,3.5) to[D,l=$VD$] (3.5,3.5);
  \draw (3.5,3.5) to[battery,l=6V] (3.5,0);
  \draw (3.5,0) -- (0,0);
  \draw (1.5,4.2) node[font=\small] {(a)};
\end{circuitikz}
